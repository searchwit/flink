CREATE OR REPLACE VIEW "V_CC_REF_LISTVALUE" ("REF_TYPE", "REF_VAL", "REF_LABEL") AS
select 'countryCodeType', country_code, country_name from cc_country
union all
select 'stateCodeType', country_code || state_id as state_ccid, state_name from cc_state
union all
select 'ccSpecialListType', 'ignore_name' as list_type, 'ignore_name' as list_type_name from dual
union all
select 'ccSpecialListType', 'boundary_word' as list_type, 'boundary_word' as list_type_name from dual
union all
select 'ccSpecialListType', 'common_place_name' as list_type, 'common_place_name' as list_type_name from dual
;
